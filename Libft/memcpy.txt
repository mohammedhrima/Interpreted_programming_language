func memcpy(src, dest):
    i = 0
    output("src len is: ", src.len, "\n")
    while i < src.len and i < dest.len:
        dest[i] = src[i]
        output("i is: ", i, "\n")
        i += 1
    if i < src.len:
        if src[i].type == dest.type:
            output("i is: ", i, "\n")
            dest = dest + src[i]
    return dest
